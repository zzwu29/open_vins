66  void feed_imu(const ov_core::ImuData &message, double oldest_time = -1) {
67 
68  // Append it to our vector
69  std::lock_guard<std::mutex> lck(imu_data_mtx);
70  imu_data.emplace_back(message);
71 
72  // Loop through and delete imu messages that are older than our requested time
73  if (oldest_time != -1) {
74  auto it0 = imu_data.begin();
75  while (it0 != imu_data.end()) {
76  if (it0->timestamp < oldest_time - 0.10) {
77  it0 = imu_data.erase(it0);
78  } else {
79  it0++;
80  }
81  }
82  }
83  }

141  static ov_core::ImuData interpolate_data(const ov_core::ImuData &imu_1, const ov_core::ImuData &imu_2, double timestamp) {
142  // time-distance lambda
143  double lambda = (timestamp - imu_1.timestamp) / (imu_2.timestamp - imu_1.timestamp);
144  // PRINT_DEBUG("lambda - %d\n", lambda);
145  // interpolate between the two times
146  ov_core::ImuData data;
147  data.timestamp = timestamp;
148  data.am = (1 - lambda) * imu_1.am + lambda * imu_2.am;
149  data.wm = (1 - lambda) * imu_1.wm + lambda * imu_2.wm;
150  return data;
151  }
