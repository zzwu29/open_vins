353  template <class T> void parse(const cv::FileNode &file_node, const std::string &node_name, T &node_result, bool required = true) {
354 
355  // Check that we have the requested node
356  if (!node_found(file_node, node_name)) {
357  if (required) {
358  PRINT_WARNING(YELLOW "the node %s of type [%s] was not found...\n" RESET, node_name.c_str(), typeid(node_result).name());
359  all_params_found_successfully = false;
360  } else {
361  PRINT_DEBUG("the node %s of type [%s] was not found (not required)...\n", node_name.c_str(), typeid(node_result).name());
362  }
363  return;
364  }
365 
366  // Now try to get it from the config
367  try {
368  file_node[node_name] >> node_result;
369  } catch (...) {
370  if (required) {
371  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in the config file!\n" RESET, node_name.c_str(),
372  typeid(node_result).name());
373  all_params_found_successfully = false;
374  } else {
375  PRINT_DEBUG("unable to parse %s node of type [%s] in the config file (not required)\n", node_name.c_str(),
376  typeid(node_result).name());
377  }

388  void parse(const cv::FileNode &file_node, const std::string &node_name, bool &node_result, bool required = true) {
389 
390  // Check that we have the requested node
391  if (!node_found(file_node, node_name)) {
392  if (required) {
393  PRINT_WARNING(YELLOW "the node %s of type [%s] was not found...\n" RESET, node_name.c_str(), typeid(node_result).name());
394  all_params_found_successfully = false;
395  } else {
396  PRINT_DEBUG("the node %s of type [%s] was not found (not required)...\n", node_name.c_str(), typeid(node_result).name());
397  }
398  return;
399  }
400 
401  // Now try to get it from the config
402  try {
403  if (file_node[node_name].isInt() && (int)file_node[node_name] == 1) {
404  node_result = true;
405  return;
406  }
407  if (file_node[node_name].isInt() && (int)file_node[node_name] == 0) {
408  node_result = false;
409  return;
410  }
411  // NOTE: we select the first bit of text as there can be a comment afterwards
412  // NOTE: for example we could have "key: true #comment here..." where we only want "true"
413  std::string value;
414  file_node[node_name] >> value;
415  value = value.substr(0, value.find_first_of('#'));
416  value = value.substr(0, value.find_first_of(' '));
417  if (value == "1" || value == "true" || value == "True" || value == "TRUE") {
418  node_result = true;
419  } else if (value == "0" || value == "false" || value == "False" || value == "FALSE") {
420  node_result = false;
421  } else {
422  PRINT_WARNING(YELLOW "the node %s has an invalid boolean type of [%s]\n" RESET, node_name.c_str(), value.c_str());
423  all_params_found_successfully = false;
424  }
425  } catch (...) {
426  if (required) {
427  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in the config file!\n" RESET, node_name.c_str(),
428  typeid(node_result).name());
429  all_params_found_successfully = false;
430  } else {
431  PRINT_DEBUG("unable to parse %s node of type [%s] in the config file (not required)\n", node_name.c_str(),
432  typeid(node_result).name());
433  }
434  }
435  }

444  void parse(const cv::FileNode &file_node, const std::string &node_name, Eigen::Matrix3d &node_result, bool required = true) {
445 
446  // Check that we have the requested node
447  if (!node_found(file_node, node_name)) {
448  if (required) {
449  PRINT_WARNING(YELLOW "the node %s of type [%s] was not found...\n" RESET, node_name.c_str(), typeid(node_result).name());
450  all_params_found_successfully = false;
451  } else {

453  }
454  return;
455  }
456 
457  // Now try to get it from the config
458  node_result = Eigen::Matrix3d::Identity();
459  try {

464  }
465  } catch (...) {
466  if (required) {
467  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in the config file!\n" RESET, node_name.c_str(),
468  typeid(node_result).name());
469  all_params_found_successfully = false;
470  } else {
471  PRINT_DEBUG("unable to parse %s node of type [%s] in the config file (not required)\n", node_name.c_str(),

484  void parse(const cv::FileNode &file_node, const std::string &node_name, Eigen::Matrix4d &node_result, bool required = true) {
485 
486  // See if we need to flip the node name
487  std::string node_name_local = node_name;
488  if (node_name == "T_cam_imu" && !node_found(file_node, node_name)) {
489  PRINT_INFO("parameter T_cam_imu not found, trying T_imu_cam instead (will return T_cam_imu still)!\n");
490  node_name_local = "T_imu_cam";
491  } else if (node_name == "T_imu_cam" && !node_found(file_node, node_name)) {
492  PRINT_INFO("parameter T_imu_cam not found, trying T_cam_imu instead (will return T_imu_cam still)!\n");
493  node_name_local = "T_cam_imu";
494  }
495 
496  // Check that we have the requested node
497  if (!node_found(file_node, node_name_local)) {
498  if (required) {
499  PRINT_WARNING(YELLOW "the node %s of type [%s] was not found...\n" RESET, node_name_local.c_str(), typeid(node_result).name());
500  all_params_found_successfully = false;
501  } else {
502  PRINT_DEBUG("the node %s of type [%s] was not found (not required)...\n", node_name_local.c_str(), typeid(node_result).name());
503  }
504  return;
505  }
506 
507  // Now try to get it from the config
508  node_result = Eigen::Matrix4d::Identity();
509  try {
510  for (int r = 0; r < (int)file_node[node_name_local].size() && r < 4; r++) {
511  for (int c = 0; c < (int)file_node[node_name_local][r].size() && c < 4; c++) {
512  node_result(r, c) = (double)file_node[node_name_local][r][c];
513  }
514  }
515  } catch (...) {
516  if (required) {
517  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in the config file!\n" RESET, node_name.c_str(),
518  typeid(node_result).name());
519  all_params_found_successfully = false;
520  } else {
521  PRINT_DEBUG("unable to parse %s node of type [%s] in the config file (not required)\n", node_name.c_str(),
522  typeid(node_result).name());
523  }
524  }
525 
526  // Finally if we flipped the transform, get the correct value
527  if (node_name_local != node_name) {
528  Eigen::Matrix4d tmp(node_result);
529  node_result = ov_core::Inv_se3(tmp);
530  }

544  template <class T> void parse_config_yaml(const std::string &node_name, T &node_result, bool required = true) {
545 
546  // Directly return if the config hasn't been opened
547  if (config == nullptr)
548  return;
549 
550  // Else lets get the one from the config
551  try {
552  parse(config->root(), node_name, node_result, required);
553  } catch (...) {
554  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in the config file!\n" RESET, node_name.c_str(),
555  typeid(node_result).name());
556  all_params_found_successfully = false;
557  }

575  template <class T>
576  void parse_external_yaml(const std::string &external_node_name, const std::string &sensor_name, const std::string &node_name,
577  T &node_result, bool required = true) {
578  // Directly return if the config hasn't been opened
579  if (config == nullptr)
580  return;
581 
582  // Create the path the external yaml file
583  std::string path;
584  if (!node_found(config->root(), external_node_name)) {
585  PRINT_ERROR(RED "the external node %s could not be found!\n" RESET, external_node_name.c_str());
586  std::exit(EXIT_FAILURE);
587  }
588  (*config)[external_node_name] >> path;
589  std::string relative_folder = config_path_.substr(0, config_path_.find_last_of('/')) + "/";
590 
591  // Now actually try to load them from file!
592  auto config_external = std::make_shared<cv::FileStorage>(relative_folder + path, cv::FileStorage::READ);
593  if (!config_external->isOpened()) {
594  PRINT_ERROR(RED "unable to open the configuration file!\n%s\n" RESET, (relative_folder + path).c_str());
595  std::exit(EXIT_FAILURE);
596  }
597 
598  // Check that we have the requested node
599  if (!node_found(config_external->root(), sensor_name)) {
600  PRINT_WARNING(YELLOW "the sensor %s of type [%s] was not found...\n" RESET, sensor_name.c_str(), typeid(node_result).name());
601  all_params_found_successfully = false;
602  return;
603  }
604 
605  // Else lets get it!
606  try {
607  parse((*config_external)[sensor_name], node_name, node_result, required);
608  } catch (...) {
609  PRINT_WARNING(YELLOW "unable to parse %s node of type [%s] in [%s] in the external %s config file!\n" RESET, node_name.c_str(),
610  typeid(node_result).name(), sensor_name.c_str(), external_node_name.c_str());
611  all_params_found_successfully = false;
612  }
613  }
