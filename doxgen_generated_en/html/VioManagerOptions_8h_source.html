154  void print_and_load_noise(const std::shared_ptr<ov_core::YamlParser> &parser = nullptr) {
155  PRINT_DEBUG("NOISE PARAMETERS:\n");
156  if (parser != nullptr) {
157  parser->parse_external("relative_config_imu", "imu0", "gyroscope_noise_density", imu_noises.sigma_w);
158  parser->parse_external("relative_config_imu", "imu0", "gyroscope_random_walk", imu_noises.sigma_wb);
159  parser->parse_external("relative_config_imu", "imu0", "accelerometer_noise_density", imu_noises.sigma_a);

162  imu_noises.sigma_wb_2 = std::pow(imu_noises.sigma_wb, 2);
163  imu_noises.sigma_a_2 = std::pow(imu_noises.sigma_a, 2);
164  imu_noises.sigma_ab_2 = std::pow(imu_noises.sigma_ab, 2);
165  }

169  parser->parse_config("up_msckf_chi2_multipler", msckf_options.chi2_multipler);
170  parser->parse_config("up_slam_sigma_px", slam_options.sigma_pix);

215  void print_and_load_state(const std::shared_ptr<ov_core::YamlParser> &parser = nullptr) {
216  if (parser != nullptr) {
217  parser->parse_config("gravity_mag", gravity_mag);
218  parser->parse_config("max_cameras", state_options.num_cameras); // might be redundant
219  parser->parse_config("downsample_cameras", downsample_cameras); // might be redundant
220  for (int i = 0; i < state_options.num_cameras; i++) {
221 
222  // Time offset (use the first one)
223  // TODO: support multiple time offsets between cameras
224  if (i == 0) {
225  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "timeshift_cam_imu", calib_camimu_dt, false);
226  }
227 
228  // Distortion model
229  std::string dist_model = "radtan";
230  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "distortion_model", dist_model);
231 
232  // Distortion parameters
233  std::vector<double> cam_calib1 = {1, 1, 0, 0};
234  std::vector<double> cam_calib2 = {0, 0, 0, 0};
235  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "intrinsics", cam_calib1);
236  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "distortion_coeffs", cam_calib2);
237  Eigen::VectorXd cam_calib = Eigen::VectorXd::Zero(8);
238  cam_calib << cam_calib1.at(0), cam_calib1.at(1), cam_calib1.at(2), cam_calib1.at(3), cam_calib2.at(0), cam_calib2.at(1),
239  cam_calib2.at(2), cam_calib2.at(3);
240  cam_calib(0) /= (downsample_cameras) ? 2.0 : 1.0;
241  cam_calib(1) /= (downsample_cameras) ? 2.0 : 1.0;
242  cam_calib(2) /= (downsample_cameras) ? 2.0 : 1.0;
243  cam_calib(3) /= (downsample_cameras) ? 2.0 : 1.0;
244 
245  // FOV / resolution
246  std::vector<int> matrix_wh = {1, 1};
247  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "resolution", matrix_wh);
248  matrix_wh.at(0) /= (downsample_cameras) ? 2.0 : 1.0;
249  matrix_wh.at(1) /= (downsample_cameras) ? 2.0 : 1.0;
250  std::pair<int, int> wh(matrix_wh.at(0), matrix_wh.at(1));
251 
252  // Extrinsics
253  Eigen::Matrix4d T_CtoI = Eigen::Matrix4d::Identity();
254  parser->parse_external("relative_config_imucam", "cam" + std::to_string(i), "T_imu_cam", T_CtoI);
255 
256  // Load these into our state
257  Eigen::Matrix<double, 7, 1> cam_eigen;
258  cam_eigen.block(0, 0, 4, 1) = ov_core::rot_2_quat(T_CtoI.block(0, 0, 3, 3).transpose());
259  cam_eigen.block(4, 0, 3, 1) = -T_CtoI.block(0, 0, 3, 3).transpose() * T_CtoI.block(0, 3, 3, 1);
260 
261  // Create intrinsics model
262  if (dist_model == "equidistant") {
263  camera_intrinsics.insert({i, std::make_shared<ov_core::CamEqui>(matrix_wh.at(0), matrix_wh.at(1))});
264  camera_intrinsics.at(i)->set_value(cam_calib);
265  } else {
266  camera_intrinsics.insert({i, std::make_shared<ov_core::CamRadtan>(matrix_wh.at(0), matrix_wh.at(1))});
267  camera_intrinsics.at(i)->set_value(cam_calib);
268  }
269  camera_extrinsics.insert({i, cam_eigen});
270  }
271  parser->parse_config("use_mask", use_mask);
272  if (use_mask) {
273  for (int i = 0; i < state_options.num_cameras; i++) {
274  std::string mask_path;
275  std::string mask_node = "mask" + std::to_string(i);
276  parser->parse_config(mask_node, mask_path);
277  std::string total_mask_path = parser->get_config_folder() + mask_path;
278  if (!boost::filesystem::exists(total_mask_path)) {
279  PRINT_ERROR(RED "VioManager(): invalid mask path:\n" RESET);
280  PRINT_ERROR(RED "\t- mask%d - %s\n" RESET, i, total_mask_path.c_str());
281  std::exit(EXIT_FAILURE);
282  }
283  masks.insert({i, cv::imread(total_mask_path, cv::IMREAD_GRAYSCALE)});
284  }
285  }
286  }
287  PRINT_DEBUG("STATE PARAMETERS:\n");
288  PRINT_DEBUG(" - gravity_mag: %.4f\n", gravity_mag);
289  PRINT_DEBUG(" - gravity: %.3f, %.3f, %.3f\n", 0.0, 0.0, gravity_mag);
290  PRINT_DEBUG(" - camera masks?: %d\n", use_mask);
291  if (state_options.num_cameras != (int)camera_intrinsics.size() || state_options.num_cameras != (int)camera_extrinsics.size()) {
292  PRINT_ERROR(RED "[SIM]: camera calib size does not match max cameras...\n" RESET);
293  PRINT_ERROR(RED "[SIM]: got %d but expected %d max cameras (camera_intrinsics)\n" RESET, (int)camera_intrinsics.size(),
294  state_options.num_cameras);
295  PRINT_ERROR(RED "[SIM]: got %d but expected %d max cameras (camera_extrinsics)\n" RESET, (int)camera_extrinsics.size(),
296  state_options.num_cameras);
297  std::exit(EXIT_FAILURE);
298  }
299  PRINT_DEBUG(" - calib_camimu_dt: %.4f\n", calib_camimu_dt);
300  for (int n = 0; n < state_options.num_cameras; n++) {
301  std::stringstream ss;
302  ss << "cam_" << n << "_fisheye:" << (std::dynamic_pointer_cast<ov_core::CamEqui>(camera_intrinsics.at(n)) != nullptr) << std::endl;
303  ss << "cam_" << n << "_wh:" << std::endl << camera_intrinsics.at(n)->w() << " x " << camera_intrinsics.at(n)->h() << std::endl;
304  ss << "cam_" << n << "_intrinsic(0:3):" << std::endl
305  << camera_intrinsics.at(n)->get_value().block(0, 0, 4, 1).transpose() << std::endl;
306  ss << "cam_" << n << "_intrinsic(4:7):" << std::endl
307  << camera_intrinsics.at(n)->get_value().block(4, 0, 4, 1).transpose() << std::endl;
308  ss << "cam_" << n << "_extrinsic(0:3):" << std::endl << camera_extrinsics.at(n).block(0, 0, 4, 1).transpose() << std::endl;
309  ss << "cam_" << n << "_extrinsic(4:6):" << std::endl << camera_extrinsics.at(n).block(4, 0, 3, 1).transpose() << std::endl;
310  Eigen::Matrix4d T_CtoI = Eigen::Matrix4d::Identity();
311  T_CtoI.block(0, 0, 3, 3) = ov_core::quat_2_Rot(camera_extrinsics.at(n).block(0, 0, 4, 1)).transpose();
312  T_CtoI.block(0, 3, 3, 1) = -T_CtoI.block(0, 0, 3, 3) * camera_extrinsics.at(n).block(4, 0, 3, 1);
313  ss << "T_C" << n << "toI:" << std::endl << T_CtoI << std::endl << std::endl;
314  PRINT_DEBUG(ss.str().c_str());
315  }
316  }
